window.onload = getData();
function getData() {
    var requestURL = 'https://byui-cit230.github.io/weather/data/towndata.json';
        var request = new XMLHttpRequest();
        request.open('GET', requestURL);
        request.responseType = 'json';
        request.send();
  
    request.onload = function() {
      var events = request.response;
      display(events);
    }
}
function display(events) {
    var eventDiv = document.getElementById("event");
    var eventText = document.createElement("p");

   if (document.title.match("Preston")) {
    var array = 4;
   } else if (document.title.match("Soda Springs")) {
    var array = 5;
   } else if (document.title.match("Fish Haven")) {
    var array = 1;
   }
 
    for (let x = 0; x < events.towns[array].events.length; x++) {
        var event = events.towns[array].events[x];
        eventText.innerHTML += event + "<br>";
    }
    eventDiv.appendChild(eventText);

}
